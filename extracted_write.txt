<script>
    let aiSessionId = '@Guid.NewGuid().ToString()';
    
    // Tự động kích hoạt bóng thoại và hoạt ảnh nảy sau 5 giây nếu người dùng không tương tác
    document.addEventListener("DOMContentLoaded", function() {
        setTimeout(function() {
            const bubble = document.getElementById('aiChatBubble');
            const botBtn = document.getElementById('aiChatBtn');
            const widget = document.getElementById('aiChatWidget');
            
            // Chỉ hiện bubble nếu khung chat đang đóng
            if (bubble && widget && widget.style.display === 'none') {
                bubble.style.display = 'block';
                
                // Kích hoạt hoạt ảnh nhảy nảy (bounce)
                if (botBtn) {
                    botBtn.classList.add('bot-bounce-alert');
                    botBtn.style.animation = 'none'; // Tắt animation float mặc định
                    botBtn.offsetHeight; // Trigger reflow
                    botBtn.style.animation = 'bounce-animation 1.2s ease 2'; // Nhảy nảy 2 lần rồi dừng
                    
                    // Sau khi nhảy xong 2.4 giây, trả về float animation bình thường
                    setTimeout(function() {
                        botBtn.classList.remove('bot-bounce-alert');
                        botBtn.style.animation = 'float-animation 3s ease-in-out infinite';
                    }, 2400);
                }
            }
        }, 5000);
    });

    function closeAiBubble() {
        const bubble = document.getElementById('aiChatBubble');
        if (bubble) bubble.style.display = 'none';
    }

    function toggleAIChat() {
        const widget = document.getElementById('aiChatWidget');
        const container = document.getElementById('aiChatBtnContainer');
        const bubble = document.getElementById('aiChatBubble');
        
        if (widget.style.display === 'none' || widget.style.display === '') {
            widget.style.display = 'block';
            if (container) {
                container.classList.add('d-none'); // Ẩn triệt để bằng d-none
            }
            if (bubble) bubble.style.display = 'none';
            document.getElementById('aiChatInput').focus();
        } else {
            widget.style.display = 'none';
            if (container) {
                container.classList.remove('d-none'); // Hiện lại khi đóng widget
            }
        }
    }

    function sendAIChat() {
        const input = document.getElementById('aiChatInput');
        const msg = input.value.trim();
        if (!msg) return;

        input.value = '';
        appendUserMessage(msg);
        showTypingIndicator();

        fetch('/api/ai-assistant/chat', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ message: msg, sessionId: aiSessionId })
        })
        .then(res => res.json())
        .then(data => {
            removeTypingIndicator();
            if (data.success) {
                appendBotMessage(data.data);
            } else {
                appendBotError("Xin lỗi, hệ thống AI đang bảo trì.");
            }
        })
        .catch(err => {
            removeTypingIndicator();
            appendBotError("Lỗi kết nối đến máy chủ AI.");
        });
    }

    function appendUserMessage(msg) {
        const body = document.getElementById('aiChatBody');
        body.innerHTML += `
            <div class="d-flex mb-3 justify-content-end">
                <div class="bg-white border text-dark p-3 rounded-4 shadow-sm" style="max-width: 80%; border-top-right-radius: 0; font-size: 0.85rem; line-height: 1.4;">
                    ${msg}
                </div>
            </div>`;
        body.scrollTop = body.scrollHeight;
    }

    function showTypingIndicator() {
        const body = document.getElementById('aiChatBody');
        body.innerHTML += `
            <div class="d-flex mb-3 gap-2" id="typingIndicator">
                <div class="rounded-circle bg-white shadow-sm p-1 d-flex align-items-center justify-content-center flex-shrink-0" style="width: 30px; height: 30px;">
                    <img src="/images/ai_helper_robot.png" alt="AI" style="width: 20px; height: 20px; object-fit: contain;" />
                </div>
                <div class="bg-white border p-3 rounded-4 shadow-sm" style="max-width: 80%; border-top-left-radius: 0;">
                    <div class="ai-typing"></div><div class="ai-typing"></div><div class="ai-typing"></div>
                </div>
            </div>`;
        body.scrollTop = body.scrollHeight;
    }

    function removeTypingIndicator() {
        const typing = document.getElementById('typingIndicator');
        if (typing) typing.remove();
    }

    function appendBotError(msg) {
        const body = document.getElementById('aiChatBody');
        body.innerHTML += `
            <div class="d-flex mb-3 gap-2">
                <div class="rounded-circle bg-white shadow-sm p-1 d-flex align-items-center justify-content-center flex-shrink-0" style="width: 30px; height: 30px;">
                    <img src="/images/ai_helper_robot.png" alt="AI" style="width: 20px; height: 20px; object-fit: contain;" />
                </div>
                <div class="bg-danger text-white p-3 rounded-4 shadow-sm" style="max-width: 80%; border-top-left-radius: 0; font-size: 0.85rem;">
                    ${msg}
                </div>
            </div>`;
        body.scrollTop = body.scrollHeight;
    }

    function appendBotMessage(data) {
        const body = document.getElementById('aiChatBody');
        
        let pillsHtml = '';
        if (data.extractedFilters) {
            const f = data.extractedFilters;
            if (f.location) pillsHtml += `<span class="badge bg-info text-dark me-1 mb-1" style="font-size: 0.75rem;"><i class="fa-solid fa-location-dot me-1"></i>${f.location}</span>`;
            if (f.landmark) pillsHtml += `<span class="badge bg-warning text-dark me-1 mb-1" style="font-size: 0.75rem;"><i class="fa-solid fa-map-pin me-1"></i>${f.landmark}</span>`;
            if (f.roomType) pillsHtml += `<span class="badge bg-secondary me-1 mb-1" style="font-size: 0.75rem;"><i class="fa-solid fa-house me-1"></i>${f.roomType}</span>`;
            if (f.maxPrice) pillsHtml += `<span class="badge bg-success me-1 mb-1" style="font-size: 0.75rem;"><i class="fa-solid fa-money-bill me-1"></i><= ${f.maxPrice.toLocaleString('vi-VN')}</span>`;
            if (f.amenities && f.amenities.length > 0) {
                f.amenities.forEach(a => {
                    pillsHtml += `<span class="badge bg-primary bg-opacity-75 me-1 mb-1" style="font-size: 0.75rem;"><i class="fa-solid fa-check me-1"></i>${a}</span>`;
                });
            }
        }

        let msgHtml = `
            <div class="d-flex mb-3 gap-2">
                <div class="rounded-circle bg-white shadow-sm p-1 d-flex align-items-center justify-content-center flex-shrink-0" style="width: 30px; height: 30px;">
                    <img src="/images/ai_helper_robot.png" alt="AI" style="width: 20px; height: 20px; object-fit: contain;" />
                </div>
                <div class="bg-white border text-dark p-3 rounded-4 shadow-sm w-100" style="max-width: 80%; border-top-left-radius: 0; font-size: 0.85rem; line-height: 1.4;">
                    <div class="mb-1">${data.replyText}</div>
                    ${pillsHtml ? `<div class="mt-2 pt-2 border-top border-light">${pillsHtml}</div>` : ''}
                </div>
            </div>`;

        if (data.suggestedRooms && data.suggestedRooms.length > 0) {
            let roomsHtml = '<div class="d-flex flex-column gap-2 mb-3 ms-5">';
            data.suggestedRooms.forEach(r => {
                roomsHtml += `
                    <a href="/Home/Detail/${r.id}" class="text-decoration-none text-dark">
                        <div class="card chat-room-card border-0 shadow-sm" style="transition: all 0.2s; border-radius: 12px; overflow: hidden;">
                            <div class="row g-0">
                                <div class="col-4">
                                    <img src="${r.mainImageUrl}" class="img-fluid h-100 object-fit-cover" style="min-height: 70px;" alt="...">
                                </div>
                                <div class="col-8">
                                    <div class="card-body p-2">
                                        <h6 class="card-title text-truncate small fw-bold mb-1">${r.title}</h6>
                                        <p class="card-text text-danger small fw-bold mb-0">${r.price.toLocaleString('vi-VN')} đ</p>
                                        <p class="card-text text-muted" style="font-size: 0.7rem;"><i class="fa-solid fa-ruler-combined me-1"></i>${r.area} m²</p>
                                    </div>
                                </div>
                            </div>
                        </div>
                    </a>`;
            });
            roomsHtml += '</div>';
            msgHtml += roomsHtml;
        }

        body.innerHTML += msgHtml;
        body.scrollTop = body.scrollHeight;
    }

    function triggerBannerAISearch() {
        const bannerInput = document.getElementById('aiBannerInput');
        const query = bannerInput.value.trim();
        if (!query) return;

        bannerInput.value = '';

        const widget = document.getElementById('aiChatWidget');
        const container = document.getElementById('aiChatBtnContainer');
        if (widget.style.display === 'none') {
            widget.style.display = 'block';
            if (container) {
                container.classList.add('d-none');
            }
        }

        const chatInput = document.getElementById('aiChatInput');
        chatInput.value = query;
        sendAIChat();
    }
</script>
